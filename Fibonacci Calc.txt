int main()
{

	int n;
	int Fibunacci;
	int Fibuminus1;

	else
	{
		if (n == 1) printf("Fibunacci = 0 1\n");
		else
		{
			Fibuminus1 = 1;
			Fibuminus2 = 0;
			printf("Fibunaccifolge:\n%d\n%d\n", Fibuminus2, Fibuminus1);

			{
				printf("Anzahl der Additionen = %d\n", i);

			}

	}
